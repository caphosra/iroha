void markAs(IrohaOrderStatus status, DateTime time) {
  switch (status) {
    case IrohaOrderStatus.POSTED:
      posted = time;
      break;
    case IrohaOrderStatus.COOKED:
      cooked = time;
      break;
    case IrohaOrderStatus.SERVED:
      served = time;
      break;
    case IrohaOrderStatus.PAID:
      paid = time;
      break;
  }
}
